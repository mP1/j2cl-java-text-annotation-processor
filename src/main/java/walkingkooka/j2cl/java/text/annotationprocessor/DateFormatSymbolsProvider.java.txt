package walkingkooka.j2cl.java.text;

final class DateFormatSymbolsProvider {

  public final static String ANNOTATION_PROCESSOR_LOCALES_FILTER = $FILTERED_LOCALES;

  public final static String SELECTED_LOCALES = $SELECTED_LOCALES;

$REGISTER_METHOD

   private DateFormatSymbolsProvider(final String locales,
                                     final String ampms,
                                     final String eras,
                                     final String months,
                                     final String shortMonths,
                                     final String shortWeekdays,
                                     final String weekdays) {
        super();
        this.locales = locales;
        this.ampms = ampms;
        this.eras = eras;
        this.months = months;
        this.shortMonths = shortMonths;
        this.shortWeekdays = shortWeekdays;
        this.weekdays = weekdays;
    }

    final String locales;
    final String ampms;
    final String eras;
    final String months;
    final String shortMonths;
    final String shortWeekdays;
    final String weekdays;
}
